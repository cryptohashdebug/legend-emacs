#+BEGIN_SRC emacs-lisp
  (defun my/font-available-p (font-name)
    "Devuelve t si FONT-NAME esta instalada en el sistema."
    (find-font (font-spec :name font-name)))

  (defun my/install-font (package-name-debian package-name-arch package-name-fedora)
    "Instala un paquete de fuente usando apt-get, pacman o dnf segÃºn la distro."
    (cond
     ((my/debian-based-p)
      (shell-command
       (format "sudo apt-get install -y %s" package-name-debian)))
     ((my/arch-based-p)
      (shell-command
       (format "sudo pacman -S --noconfirm %s" package-name-arch)))
     ((my/fedora-based-p)
      (shell-command
       (format "sudo dnf install -y %s" package-name-fedora)))))
  
  (defun my/install-font-with-apt (package-name)
  "Instala un paquete de fuente usando apt-get."
  (shell-command
   (format "sudo apt-get install -y %s" package-name)))
#+END_SRC

  (defun my/debian-based-p ()
    "Devuelve t si el sistema parece ser Debian/Ubuntu."
    (and (eq system-type 'gnu/linux)
         (file-exists-p "/etc/debian_version")))

  (defun my/arch-based-p ()
    "Devuelve t si el sistema parece ser Arch Linux."
    (and (eq system-type 'gnu/linux)
         (file-exists-p "/etc/arch-release")))

  (defun my/fedora-based-p ()
    "Devuelve t si el sistema parece ser Fedora."
    (and (eq system-type 'gnu/linux)
         (file-exists-p "/etc/fedora-release")))

#+BEGIN_SRC emacs-lisp
  (load-theme 'tango-dark t)         ;; Tema oscuro

  ;; Configuración de fuente con verificación e instalación
  (let ((preferred-font "Fira Code Retina")
        (preferred-package "fonts-firacode")
        (fallback-font "DejaVu Sans Mono"))
    (if (my/font-available-p preferred-font)
        ;; Si la fuente esta disponible, usarla
        (set-face-attribute 'default nil :font preferred-font :height 120)
      ;; Si no esta disponible, intentar instalarla y usar fallback
      (progn
        (message "La fuente %s no esta instalada. Intentando instalar..." preferred-font)
        (my/install-font-with-apt preferred-package)
        (if (my/font-available-p preferred-font)
            (set-face-attribute 'default nil :font preferred-font :height 120)
          (set-face-attribute 'default nil :font fallback-font :height 120)))))

    ;; Si ninguna fuente se pudo usar, aplicar fallback generico
    (unless (find-font (font-spec :name "DejaVu Sans Mono"))
      (set-face-attribute 'default nil :font "Monospace" :height 120)
      (message "Usando fallback genérico: Monospace"))

      (use-package dashboard
        :ensure t
        :defer nil
        :preface
        (defun update-config ()
          "Actualizando Legend a la ultima version, desde el Dashboard"
          (interactive)
          (let ((dir (expand-file-name user-emacs-directory)))
      	(if (file-exists-p dir)
      	    (progn
      	      (message "Legend esta actualizando!!")
      	      (cd dir)
      	      (shell-command "git pull")
      	      (message "Actualizacion terminada. Vea los cambios en el buffer de mensajes y reinicie Emacs"))
      	  (message "\"%s\" doesn't exist." dir))))

        (defun create-scratch-buffer ()
          "Creado u buffer scratch, se usa en el Dashboard"
          (interactive)
          (switch-to-buffer (get-buffer-create "*scratch*"))
          (lisp-interaction-mode))
        :config

        (dashboard-setup-startup-hook)
        (setq dashboard-items '((recents . 5)))
        (setq dashboard-startupify-list
      	    '(dashboard-insert-banner
